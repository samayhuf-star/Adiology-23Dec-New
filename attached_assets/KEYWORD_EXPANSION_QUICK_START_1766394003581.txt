function expandKeywords(seedKeywords, matchTypes, country) {
  const expanded = new Set()

  for (const seed of seedKeywords) {
    // Add original
    expanded.add(seed)

    // Add modifier variations
    const modifiers = getModifiers(country)
    for (const mod of modifiers) {
      expanded.add(`${mod} ${seed}`)
    }

    // Add location variations
    const locations = getLocations(country)
    for (const loc of locations) {
      expanded.add(`${seed} ${loc}`)
    }

    // Add questions
    const questions = generateQuestions(seed)
    for (const q of questions) {
      expanded.add(q)
    }

    // Add long-tails
    const longTails = generateLongTails(seed, modifiers, locations)
    for (const lt of longTails) {
      expanded.add(lt)
    }

    // ... more expansions
  }

  return Array.from(expanded)
}
